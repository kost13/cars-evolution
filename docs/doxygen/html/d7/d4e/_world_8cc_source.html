122  simulation_data_->reset(cars_num);

124  float timeStep = 1.0f / 60;
125 
126  // creating physical car objects
127  //'population_' was assigned in constructor
128  generateCars();
129 
130  bool stop = false; // flag for simulation control
131  int iter = 0;
132  int flag_stop = 0; // 1 if one of cars reached ending
133 
134  // used for simualtion control
135  b2Vec2 last_position;
136  b2Vec2 diff_position; // for b2vec2 only -= operator is defined
137  last_position.SetZero();
138 
139  while (!stop && iter < 30000) {
140  world_->Step(timeStep, 8, 3);
141 
142  // for each car
143  for (const auto& car : cars_) {
144  b2Vec2 position = car->getRearWheelPos();
145  float angle = car->getAngle();
146 
147  Position position_ = {position.x, position.y, -angle};
148  simulation_data_->pushPosition(car->getCarNum(), position_);
149 
150  // maximal reached distance
151  if (position_.x > car->getMaximalDistanceReached()) {
152  car->setMaximalDistanceReached(position_.x);
153  }
154 
155  // check if any car reached ending
156  // later parameters can be taken to the settings struct
157  if (position_.x > 200 && position_.y > -5) {
158  logger::info() << "Car " << car->getCarNum()
159  << "has reached the end of the track" << std::endl;
160  flag_stop = 1;
161  break;
162  }
163 
164  /* checks if car is moving in our definition and
165  * if can be counted as stopped.
166  * Definition: if length of position shift vector for the car
167  * is smaller than 'settings.minimumLength_of_vector',
168  * the car is not moving. If it is not moving in
169  * 'settings.max_car_iter' iterations, it is counted as stopped.
170  *
171  */
172  diff_position = position;
173  diff_position -= last_position; // for b2vec2 only -= operator is defined
174 
175  int pom = car->GetIterStopped();
176  if (diff_position.Length() < 3) {
177  ++pom;
178  } else {
179  pom = 0;
180  }
181  car->setIterStopped(pom);
182 
183  position = last_position;
184 
185  if (car->GetIterStopped() > 2000) { // flag up
186  car->setStopped(true);
187  }
188 
189  iter++;
190 
191  } // for each car
192 
193  if (flag_stop != 0) {
194  break;
195  }
196 
197  // Checks if any car is moving.
198  // If not, 'stop' will be left with value 1 and simualtion stopped.
199  stop = true;

201  if (!car->getStopped()) {
202  stop = false;
203  break;
204  }
205  }
